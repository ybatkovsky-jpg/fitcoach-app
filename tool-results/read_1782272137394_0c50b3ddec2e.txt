   231тЖТ   231тЖТexport const useAppStore = create<AppState>()(
   232тЖТ   232тЖТ  persist(
   233тЖТ   233тЖТ    (set, get) => ({
   234тЖТ   234тЖТ      // Navigation
   235тЖТ   235тЖТ      screen: 'onboarding' as AppScreen,
   236тЖТ   236тЖТ      setScreen: (s) => set({ screen: s }),
   237тЖТ   237тЖТ      selectedExerciseId: null,
   238тЖТ   238тЖТ      openExerciseGuide: (id) => set({ selectedExerciseId: id, screen: 'exercise_guide' }),
   239тЖТ   239тЖТ
   240тЖТ   240тЖТ      // Onboarding
   241тЖТ   241тЖТ      onboardingStep: 0,
   242тЖТ   242тЖТ      setOnboardingStep: (s) => set({ onboardingStep: s }),
   243тЖТ   243тЖТ      isOnboarded: false,
   244тЖТ   244тЖТ
   245тЖТ   245тЖТ      // Profile
   246тЖТ   246тЖТ      profile: makeDefaultProfile() as UserProfile,
   247тЖТ   247тЖТ      setProfileField: (key, value) =>
   248тЖТ   248тЖТ        set((state) => ({
   249тЖТ   249тЖТ          profile: {
   250тЖТ   250тЖТ            ...(state.profile ?? makeDefaultProfile()),
   251тЖТ   251тЖТ            [key]: value,
   252тЖТ   252тЖТ          } as UserProfile,
   253тЖТ   253тЖТ        })),
   254тЖТ   254тЖТ
   255тЖТ   255тЖТ      completeOnboarding: () => {
   256тЖТ   256тЖТ        const state = get();
   257тЖТ   257тЖТ        if (!state.profile) return;
   258тЖТ   258тЖТ        const level = calculateFitnessLevel(state.profile, state.history.length);
   259тЖТ   259тЖТ        const updatedProfile = { ...state.profile, fitnessLevel: level };
   260тЖТ   260тЖТ        set({ isOnboarded: true, profile: updatedProfile, screen: 'dashboard' });
   261тЖТ   261тЖТ        // Auto-generate first plan with scientific periodization
   262тЖТ   262тЖТ        const recentIds = getRecentExerciseIds(state.history);
   263тЖТ   263тЖТ        const plan = generateWorkout(updatedProfile, state.customExercises, state.periodizationWeek, recentIds);
   264тЖТ   264тЖТ        set({ currentPlan: plan });
   265тЖТ   265тЖТ      },
   266тЖТ   266тЖТ
   267тЖТ   267тЖТ      // Workout plan
   268тЖТ   268тЖТ      currentPlan: null,
   269тЖТ   269тЖТ      generateNewPlan: () => {
   270тЖТ   270тЖТ        const state = get();
   271тЖТ   271тЖТ        if (!state.profile) return;
   272тЖТ   272тЖТ
   273тЖТ   273тЖТ        let plan: ReturnType<typeof generateWorkout>;
   274тЖТ   274тЖТ
   275тЖТ   275тЖТ        if (state.currentPlan && state.lastFeedback && state.history.length > 0) {
   276тЖТ   276тЖТ          // ADAPTIVE PATH: use scientific adaptive multiplier with phase context
   277тЖТ   277тЖТ          const phase = (state.currentPlan as WorkoutPlan & { periodizationPhase?: PeriodizationPhase }).periodizationPhase ?? 'accumulation';
   278тЖТ   278тЖТ          const multiplier = getScientificAdaptiveMultiplier(
   279тЖТ   279тЖТ            state.lastFeedback,
   280тЖТ   280тЖТ            state.missedDays,
   281тЖТ   281тЖТ            state.consecutiveHardCount,
   282тЖТ   282тЖТ            phase,
   283тЖТ   283тЖТ          );
   284тЖТ   284тЖТ          const adapted = applyAdaptiveLoad(state.currentPlan.exercises, multiplier);
   285тЖТ   285тЖТ          plan = {
   286тЖТ   286тЖТ            ...state.currentPlan,
   287тЖТ   287тЖТ            id: `plan_${Date.now()}`,
   288тЖТ   288тЖТ            exercises: adapted,
   289тЖТ   289тЖТ            createdAt: Date.now(),
   290тЖТ   290тЖТ          };
   291тЖТ   291тЖТ        } else {
   292тЖТ   292тЖТ          // FRESH PATH: generate with periodization and exercise rotation
   293тЖТ   293тЖТ          const recentIds = getRecentExerciseIds(state.history);
   294тЖТ   294тЖТ          plan = generateWorkout(state.profile, state.customExercises, state.periodizationWeek, recentIds);
   295тЖТ   295тЖТ        }
   296тЖТ   296тЖТ
   297тЖТ   297тЖТ        set({ currentPlan: plan });
   298тЖТ   298тЖТ      },
   299тЖТ   299тЖТ
   300тЖТ   300тЖТ      // Active workout
   301тЖТ   301тЖТ      workoutSession: null,
   302тЖТ   302тЖТ      startWorkout: () => {
   303тЖТ   303тЖТ        set({ screen: 'workout_preview' });
   304тЖТ   304тЖТ      },
   305тЖТ   305тЖТ      beginWorkout: () => {
   306тЖТ   306тЖТ        const state = get();
   307тЖТ   307тЖТ        if (!state.currentPlan) return;
   308тЖТ   308тЖТ
   309тЖТ   309тЖТ        // Reset completion state but KEEP scientifically calculated sets
   310тЖТ   310тЖТ        const newExercises = state.currentPlan.exercises.map((ex) => ({
   311тЖТ   311тЖТ          ...ex,
   312тЖТ   312тЖТ          completedSets: 0,
   313тЖТ   313тЖТ          setResults: [] as { reps: number; rpe?: number }[],
   314тЖТ   314тЖТ        }));
   315тЖТ   315тЖТ
   316тЖТ   316тЖТ        // Set rest duration from first exercise
   317тЖТ   317тЖТ        const firstRest = getRestForExercise(newExercises[0]);
   318тЖТ   318тЖТ
   319тЖТ   319тЖТ        // Record start time for first exercise
   320тЖТ   320тЖТ        const firstTiming: ExerciseTiming = {
   321тЖТ   321тЖТ          exerciseConfigId: newExercises[0]?.exerciseConfigId ?? '',
   322тЖТ   322тЖТ          startedAt: Date.now(),
   323тЖТ   323тЖТ          finishedAt: Date.now(),
   324тЖТ   324тЖТ          totalRepsDone: 0,
   325тЖТ   325тЖТ          totalSeconds: 0,
   326тЖТ   326тЖТ        };
   327тЖТ   327тЖТ
   328тЖТ   328тЖТ        set({
   329тЖТ   329тЖТ          screen: 'workout',
   330тЖТ   330тЖТ          workoutSession: {
   331тЖТ   331тЖТ            ...defaultSession(firstRest),
   332тЖТ   332тЖТ            exerciseTimings: [firstTiming],
   333тЖТ   333тЖТ          },
   334тЖТ   334тЖТ          currentPlan: { ...state.currentPlan, exercises: newExercises },
   335тЖТ   335тЖТ        });
   336тЖТ   336тЖТ      },
   337тЖТ   337тЖТ      completeSet: (reps, rpe) =>
   338тЖТ   338тЖТ        set((state) => {
   339тЖТ   339тЖТ          if (!state.workoutSession || !state.currentPlan) return state;
   340тЖТ   340тЖТ          const { currentExerciseIndex, currentSet } = state.workoutSession;
   341тЖТ   341тЖТ          const exercises = [...state.currentPlan.exercises];
   342тЖТ   342тЖТ          const exercise = { ...exercises[currentExerciseIndex] };
   343тЖТ   343тЖТ          exercise.completedSets += 1;
   344тЖТ   344тЖТ          exercise.setResults = [...exercise.setResults, { reps, rpe }];
   345тЖТ   345тЖТ          exercises[currentExerciseIndex] = exercise;
   346тЖТ   346тЖТ
   347тЖТ   347тЖТ          const isLastSet = exercise.completedSets >= exercise.targetSets;
   348тЖТ   348тЖТ          const isLastExercise = currentExerciseIndex >= exercises.length - 1;
   349тЖТ   349тЖТ
   350тЖТ   350тЖТ          // Scientific rest: use exercise's rest duration
   351тЖТ   351тЖТ          const currentRest = getRestForExercise(exercise);
   352тЖТ   352тЖТ          const shouldRest = !(isLastSet && isLastExercise);
   353тЖТ   353тЖТ
   354тЖТ   354тЖТ          return {
   355тЖТ   355тЖТ            currentPlan: { ...state.currentPlan, exercises },
   356тЖТ   356тЖТ            workoutSession: {
   357тЖТ   357тЖТ              ...state.workoutSession,
   358тЖТ   358тЖТ              isResting: shouldRest,
   359тЖТ   359тЖТ              restSecondsLeft: shouldRest ? currentRest : 0,
   360тЖТ   360тЖТ              restDuration: currentRest,
   361тЖТ   361тЖТ              currentSet: isLastSet ? 0 : currentSet + 1,
   362тЖТ   362тЖТ            },
   363тЖТ   363тЖТ          };
   364тЖТ   364тЖТ        }),
   365тЖТ   365тЖТ      skipSet: () =>
   366тЖТ   366тЖТ        set((state) => {
   367тЖТ   367тЖТ          if (!state.workoutSession || !state.currentPlan) return state;
   368тЖТ   368тЖТ          const { currentExerciseIndex, currentSet } = state.workoutSession;
   369тЖТ   369тЖТ          const exercises = [...state.currentPlan.exercises];
   370тЖТ   370тЖТ          const exercise = { ...exercises[currentExerciseIndex] };
   371тЖТ   371тЖТ          exercise.setResults = [...exercise.setResults, { reps: 0 }];
   372тЖТ   372тЖТ          exercises[currentExerciseIndex] = exercise;
   373тЖТ   373тЖТ
